1. 출력

프로그램의 기본 구조

public class Main {
    // 프로그램 시작
    public static void main( String[] args ) {
        // Hello, World!를 출력
        System.out.println( "Hello World!" );
        //프로그램 종료
    }
}

public   클래스, 변수, 메소드 등을 외부에서 접근할 수 있도록 허용

class   변수 및 상수와 이를 관리하는 메서드로 구성된 객체의 구조를 선언한 설계도

//   개발자의 이해를 돕기 위한 코드에 대한 설명으로 실행시 무시

static   static 변수나 메서드는 프로그램 시작할 때 정적으로 생성되므로 프로그램 종료시까지 사용 가능

void   메서드 종료시 반환값 없이 호출했던 곳으로 되돌아가는 메서드를 의미

main   프로그램을 시작할 때 운영체제가 호출하는 메소드

String   "apple", "banana" 등을 표현하는 문자열 클래스

String[]   문자열 여러 개를 포함하는 배열 클래스

args   main 메소드가 시작할 때 전달받는 정보들

System.out   모니터와 같은 표준 출력

println   주어진 내용과 줄바꿈(new line) 문자를 출력하는 메서드
System.out.println( "Hello, World!" );
  		  Hello, World!
 
print   주어진 내용을 출력하는 메서드
System.out.print( "Hello, World!" );   Hello, World!

;   명령어 한 줄을 끝났음을 표현

주요 제어 문자

\n   줄바꿈(new line) 문자
System.out.print( "Hello, World!\n" );
  		  Hello, World!
 
\"   큰 따옴표 문자
System.out.print( "\"Hello, World!\"" );   "Hello, World!"

\\   역슬래시 문자
System.out.print( "\\Hello, World!\\" );   \Hello, World!\

\0   빈 문자로 '\0'는 문자열 끝을 나타냄
System.out.print( "Hello, World!\0" );   Hello, World!

\033   출력 설정 문자
System.out.print( "\033[31m 빨강색(글자) \033[0m" );   빨강색(글자)
System.out.print( "\033[33m 노랑색(글자) \033[0m" );   노랑색(글자)
System.out.print( "\033[32m 초록색(글자) \033[0m" );   초록색(글자)
System.out.print( "\033[34m 파랑색(글자) \033[0m" );   파랑색(글자)
System.out.print( "\033[41m 빨강색(바탕) \033[0m" );   빨강색(바탕)
System.out.print( "\033[43m 노랑색(바탕) \033[0m" );   노랑색(바탕)
System.out.print( "\033[42m 초록색(바탕) \033[0m" );   초록색(바탕)
System.out.print( "\033[44m 파랑색(바탕) \033[0m" );   파랑색(바탕)
System.out.print( "\033[01m 글자 굵게 \033[0m" );   글자 굵게
System.out.print( "123\n 456 \033[2J 789" );
// '123\n 456 '을 모두 지우고 빈 화면의 현재 커서 위치에 ' 789'를 출력
        789
System.out.print( "123\n 456 \033[2K 789" );
// 같은 줄에 있던 ' 456 '을 지우고 빈 줄의 현재 커서 위치에 ' 789'를 출력 		123
        789
System.out.print( "\033[1;5fO" );   // 1번째줄 5번째칸에 O를 출력       O
System.out.print( "\033[3;6fO" );   // 3번째줄 6번째칸에 O를 출력
※ \033[0m는 글자색이나 배경색 초기화
